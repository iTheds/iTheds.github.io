php以键名获取键值的方法

array_values($array) //返回所有键值
$result=array_reverse($input); //将数组颠倒，不保留键名
$result_keyed=array_reverse($input,true); //将数组颠倒，保留键名
array_keys($array,”blue”); //返回值为blue的键名
