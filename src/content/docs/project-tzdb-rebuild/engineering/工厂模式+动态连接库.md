工厂模式的核心思想是使用一个抽象接口定义功能模块的行为，再用一个工厂类或工厂函数来创建这行功能模块的具体实现对象。

- 接口：定义通用的功能接口。
- 实现类：不同的功能模块具体实现该接口。
- 工厂：根据指定条件生成实现类的实例。

以存储为例：

首先定义一个通用的存储引擎接口：

```cpp
// StorageEngine.h
#ifndef STORAGE_ENGINE_H
#define STORAGE_ENGINE_H

class StorageEngine {
public:
    virtual bool insert(const char* key, const char* value) = 0;
    virtual const char* query(const char* key) = 0;
    virtual ~StorageEngine() {}
};

extern "C" {
    // 动态库导出的工厂函数名必须统一规范
    StorageEngine* create_engine();
}

#endif

```

实现具体存储引擎

```cpp
// MemStorageEngine.cpp
#include "StorageEngine.h"
#include <map>
#include <string>

class MemStorageEngine : public StorageEngine {
    std::map<std::string, std::string> data_;
public:
    bool insert(const char* key, const char* value) {
        data_[key] = value;
        return true;
    }
    const char* query(const char* key) {
        std::map<std::string, std::string>::iterator it = data_.find(key);
        if(it != data_.end())
            return it->second.c_str();
        else
            return NULL;
    }
};

// 导出工厂函数
extern "C" StorageEngine* create_engine() {
    return new MemStorageEngine();
}

```

示例程序：

```cpp
// main.cpp
#include "StorageEngine.h"
#include <dlfcn.h>
#include <iostream>

typedef StorageEngine* (*create_engine_t)();

int main() {
    // 加载插件动态库
    void* handle = dlopen("./libmemengine.so", RTLD_LAZY);
    if(!handle) {
        std::cerr << "Error loading lib: " << dlerror() << std::endl;
        return 1;
    }

    // 加载工厂函数
    create_engine_t create_engine = (create_engine_t)dlsym(handle, "create_engine");
    char* error;
    if((error = dlerror()) != NULL) {
        std::cerr << "Error loading symbol: " << error << std::endl;
        dlclose(handle);
        return 1;
    }

    // 使用工厂函数创建存储引擎实例
    StorageEngine* engine = create_engine();

    // 清理资源
    delete engine;
    dlclose(handle);

    return 0;
}

```

加载多个存储引擎

```cpp
// 伪代码示例，扫描插件目录加载插件
std::vector<StorageEngine*> engines;
for(auto plugin_name : list_all_plugins("plugins/")) {
    void* handle = dlopen(plugin_name.c_str(), RTLD_LAZY);
    create_engine_t create_engine = (create_engine_t)dlsym(handle, "create_engine");
    engines.push_back(create_engine());
}
```
