### 1.1 顺序表

```C++
#define MaxSize 50 
typedef struct {
    ElemType data[MaxSize];
    int length;
}SqList;
``` 

动态分配:

    ElemType *data;
    L.data = (ELemType*)malloc(sizeof(ElemType)*InitSize);//C
    L.data = new ElemType[InitSize];//C++

基本操作。基于C++

```C++
//插入元素,第i个位置插入e
bool ListInsert(SqList &L, int i,ElemType e){
    if(i > L.length + 1 || i < 1 || L.length >= MaxSize ){
        return false;
    }
    for(int j = L.length;j >= i;j--){
        L.data[j] = L.data[j-1];
    }
    L.data[i-1] = e;
    L.length ++;
    return true;
}
//删除元素,第i个位置删除e
bool ListDelete(SqList &L, int i,ElemType e){
    if(i > L.length + 1  ){
        return false;
    }
    for(int j = i+1 ;j <= L.length ;j++){
        L.data[j] = L.data[j+1];
    }
    L.length --;
    return true;
}
//顺序查找，查找第一个等于e的元素，返回位序
int LocateElem(SqList L ,ElemType e){
    for(int j = 0 ;j <= L.length ;j++){
        if(L.data[j] == e){
            return  j+1;
        }
    }
    return 0;
}
``` 

### 1.2 链式表示

线性表的链式存储又称单链表：数据元素，后续指针。一组任意存储单元来存储线性表中的数据元素。单链表是非随机存取的存储结构，即不能直接找到表中的某个特定的元素节点，查找某个特定的结点时，需要从表头开始遍历，依次查找。

```C++
typedef struct LNode{
    ElemType data;
    struct LNode *next;
}LNode,*LinkList;
```

头插法建立单链表的算法：

```C++
LinkList List_HeadInsert(LinkList& L) {
    LNode* s; int x;
    L = (LinkList)malloc(sizeof(LNode));
    L->next = NULL;
    scanf("%d", &x);
    while (x != 9999) {//输入9999表示结束
        s = (LNode*)malloc(sizeof(LNode));
        s->data = x;
        s->next = L->next;
        L->next = s;
        scanf("%d",&x);
    }
    return L;
}
```
